-- Criar o banco de dados
CREATE DATABASE datakero;
USE datakero;

-- Tabela de usuários (Admin e User)
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL,
    tipo ENUM('admin', 'user') NOT NULL,
    ip VARCHAR(45),
    geolocalizacao VARCHAR(255),
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabela de logs (Registra alterações feitas pelos usuários)
CREATE TABLE logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ip VARCHAR(45),
    geolocalizacao VARCHAR(255),
    campo_modificado VARCHAR(255) NOT NULL,
    valor_antigo TEXT,
    valor_novo TEXT,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
);

-- Tabela de acessos públicos (Registro de acessos de usuários não logados)
CREATE TABLE acessos_publicos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ip VARCHAR(45) NOT NULL,
    geolocalizacao VARCHAR(255),
    pagina_acessada VARCHAR(255) NOT NULL,
    tempo_pagina INT NOT NULL COMMENT 'Tempo em segundos',
    tempo_total INT NOT NULL COMMENT 'Tempo total no site em segundos'
);

-- Tabela de configurações do site (Personalização das seções)
CREATE TABLE configuracoes_site (
    id INT AUTO_INCREMENT PRIMARY KEY,
    secao VARCHAR(100) NOT NULL,
    cor_principal VARCHAR(7) NOT NULL COMMENT 'Formato HEX, ex: #FFFFFF',
    fonte VARCHAR(255),
    imagem_fundo VARCHAR(255)
);

-- Tabela de serviços (Seções dinâmicas dentro da área "Serviços")
CREATE TABLE servicos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descricao TEXT,
    ordem INT NOT NULL
);

-- Tabela para armazenar arquivos de personalização (imagens e fontes)
CREATE TABLE arquivos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    tipo ENUM('imagem', 'fonte') NOT NULL,
    nome_arquivo VARCHAR(255) NOT NULL,
    caminho VARCHAR(255) NOT NULL,
    data_upload TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
);
🔹 O que esse script faz?
✅ Cria o banco de dados datakero
✅ Tabela usuarios → Armazena dados dos administradores e usuários
✅ Tabela logs → Registra alterações feitas pelos usuários (IP e localização incluídos)
✅ Tabela acessos_publicos → BI para monitorar acessos não logados
✅ Tabela configuracoes_site → Personalização do layout pelo painel admin
✅ Tabela servicos → Permite criar seções dinâmicas na página "Serviços"
✅ Tabela arquivos → Upload de imagens e fontes para personalização

Você pode rodar esse script diretamente no MySQL para criar toda a estrutura do banco de dados. Caso precise de alguma modificação, me avise! 🚀
